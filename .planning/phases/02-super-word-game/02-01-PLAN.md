```typescript
export interface SceneItem {
  readonly id: string
  readonly type: 'letter' | 'distractor'
  readonly char?: string          // only present on type: 'letter'
  readonly emoji: string
  readonly label: string
  readonly x: number              // percentage (0-100) of scene container
  readonly y: number              // percentage (0-100) of scene container
}

export interface Puzzle {
  readonly answer: string         // uppercase, e.g. 'CAT'
  readonly prompt: string         // clue text shown to player
  readonly hint: string           // hint text
  readonly hintEmoji: string
  readonly items: readonly SceneItem[]
}

export interface CollectedLetter {
  readonly char: string
  readonly sourceId: string
}

export interface DragState {
  readonly pointerId: number
  readonly sourceIndex: number
  readonly startX: number
  readonly startY: number
  ghost: HTMLElement | null        // mutable — DOM element reference (excluded from serialization)
}

export interface GameState {
  readonly currentPuzzleIndex: number
  readonly collectedLetters: readonly CollectedLetter[]
  readonly score: number
  readonly hintUsed: boolean
  readonly selectedTileIndex: number | null
  readonly dragState: DragState | null
  readonly completed: readonly boolean[]
  readonly wowMode: boolean
}
```

**src/super-word/puzzles.ts** — All 5 puzzle datasets. Import `Puzzle` from `./types.js`. Export `PUZZLES` as `readonly Puzzle[]`. Copy exact data from prototype `super-word/app.js` (lines 15-98) — same answers (CAT, SUN, FROG, STAR, BOOK), same items with ids, types, chars, emoji, labels, x/y positions. Keep `.js` extension in import path (esbuild ESM resolution).

**src/super-word/state.ts** — Pure state functions. Import `GameState`, `Puzzle` from `./types.js`. All functions take current state and return a new state object — no mutations. Export:

- `createInitialState(puzzleCount: number, wowMode: boolean): GameState` — Returns `{ currentPuzzleIndex: 0, collectedLetters: [], score: 0, hintUsed: false, selectedTileIndex: null, dragState: null, completed: Array(puzzleCount).fill(false), wowMode }`

- `collectLetter(state: GameState, char: string, sourceId: string): GameState` — Appends `{ char, sourceId }` to collectedLetters. Returns new state with `selectedTileIndex: null`.

- `swapLetters(state: GameState, indexA: number, indexB: number): GameState` — Swaps collectedLetters at indexA and indexB. Returns new state with `selectedTileIndex: null`.

- `selectTile(state: GameState, index: number): GameState` — If `selectedTileIndex === null`, set it to index. If `selectedTileIndex === index`, set to null (deselect). If `selectedTileIndex !== null && !== index`, swap the two tiles and set selectedTileIndex to null (delegate to swapLetters internally).

- `checkAnswer(state: GameState, puzzle: Puzzle): { correct: boolean; newState: GameState }` — Joins collectedLetters chars and compares to puzzle.answer (uppercase). If correct: add 10 to score (5 if hintUsed), set `completed[currentPuzzleIndex] = true`. Return `{ correct, newState }`.

- `useHint(state: GameState): GameState` — Returns new state with `hintUsed: true`.

- `advancePuzzle(state: GameState): GameState` — Increments currentPuzzleIndex, resets collectedLetters to [], hintUsed to false, selectedTileIndex to null.

- `resetGame(state: GameState): GameState` — Returns `createInitialState(state.completed.length, state.wowMode)`.

All state is JSON-serializable (DragState.ghost is transient and excluded when serializing). No `class`, no `this`, no `let` for state mutation — spread operators and array methods only.
